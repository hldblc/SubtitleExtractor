## The project

A subtitle extractor with three interchangeable strategies (Strategy Pattern):
- embedded - pulls existing subtitle stream from a container (MKV/MP4) via FFmpeg
- whisper - speech-to-text via FFmpeg's built-in `whisper` audio filter (NOT a separate whisper.cpp binary)
- ocr - Tesseract OCR on cropped frames for burned-in subtitles

Output formats: SRT, VTT, TXT.

CLI is fully working. I successfully transcribed an 11-minute test video to 177 SRT cues. We are now adding a Qt 6 widgets GUI on top of the existing library so I can pick files via dialogs.

## Important technical gotchas (learned from previous session)

1. FFmpeg whisper filter colon escape. Windows paths contain ':' (drive letter). Inside FFmpeg's filter graph syntax, ':' is the option separator. Single-quote wrapping (model='C:/foo') does NOT survive FFmpeg's filter parser on Windows paths - empirically verified. The fix is \: (backslash escape). The existing src/WhisperExtractor.cpp already does this correctly - don't change it.
2. MSVC C4996 warnings on sscanf are silenced via add_compile_definitions(_CRT_SECURE_NO_WARNINGS) in CMakeLists.txt. Standard sscanf is portable; Microsoft's sscanf_s is not.
3. Q_OBJECT macro must appear in any class that declares signals or slots, and the class must live in a header (so AUTOMOC can find it). Forgetting Q_OBJECT produces a linker error "undefined reference to vtable for MainWindow" - that's the canonical telltale.
4. GUI extraction must run on a worker thread to avoid freezing the UI. Use QtConcurrent::run plus QMetaObject::invokeMethod(this, lambda, Qt::QueuedConnection) to marshal results back to the GUI thread.
5. windeployqt is included as a POST_BUILD step in CMakeLists.txt. It copies Qt6Widgets.dll, Qt6Core.dll, etc. next to the .exe so it can run from anywhere.

void MainWindow::onBrowseOutput() {
    QString dir = QFileDialog::getExistingDirectory(this, tr("Select output folder"));

    QString base = tr("subtitles");

        base = QFileInfo(videoPathEdit_->text()).completeBaseName();

    outputBaseEdit_->setText(dir + "/" + base);

void MainWindow::onExtractFinished(bool success, const QString& message) {
    isRunning_ = false;
    extractButton_->setEnabled(true);
    progressBar_->setVisible(false);
    statusLabel_->setText(message);
    if (success) {
        QMessageBox::information(this, tr("Done"), message);
    } else {
        QMessageBox::warning(this, tr("Extraction failed"),
                             tr("See log area for details."));
    }
}

void MainWindow::onExtractClicked() {
    if (isRunning_) return;

    if (videoPathEdit_->text().isEmpty()) {
        QMessageBox::warning(this, tr("Missing input"), tr("Please pick a video file."));
        return;
    }
    if (outputBaseEdit_->text().isEmpty()) {
        QMessageBox::warning(this, tr("Missing output"), tr("Please pick an output base path."));
        return;
    }

    const QString modeStr = modeCombo_->currentData().toString();
    if (modeStr == "whisper" && modelPathEdit_->text().isEmpty()) {
        QMessageBox::warning(this, tr("Missing model"), tr("Whisper mode needs a model (.bin) file."));
        return;
    }

    QStringList formats;
    if (srtCheck_->isChecked()) formats << "srt";
    if (vttCheck_->isChecked()) formats << "vtt";
    if (txtCheck_->isChecked()) formats << "txt";
    if (formats.isEmpty()) {
        QMessageBox::warning(this, tr("No format"), tr("Pick at least one output format."));
        return;
    }

    const std::string videoPath  = videoPathEdit_->text().toStdString();
    const std::string outputBase = outputBaseEdit_->text().toStdString();
    const std::string modelPath  = modelPathEdit_->text().toStdString();

    logArea_->clear();
    appendLog("Starting extraction...");
    appendLog(QString("Video : %1").arg(videoPathEdit_->text()));
    appendLog(QString("Mode  : %1").arg(modeStr));
    appendLog(QString("Output: %1.{%2}").arg(outputBaseEdit_->text(), formats.join("|")));

    isRunning_ = true;
    extractButton_->setEnabled(false);
    progressBar_->setVisible(true);
    statusLabel_->setText(tr("Working - this can take a few minutes..."));

    QtConcurrent::run([this, modeStr, videoPath, outputBase, modelPath, formats]() {
        try {
            FactoryOptions opts;
            opts.mode = ExtractorFactory::parseMode(modeStr.toStdString());
            opts.whisperModelPath = modelPath;

            std::unique_ptr<IExtractor> extractor = ExtractorFactory::create(opts);

            QMetaObject::invokeMethod(this, [this, name = QString::fromStdString(extractor->name())]() {
                appendLog("Strategy: " + name);
            }, Qt::QueuedConnection);

            auto entries = extractor->extract(videoPath);

            QStringList written;
            for (const auto& fmt : formats) {
                std::filesystem::path out = outputBase;
                out += ("." + fmt.toStdString());
                if      (fmt == "srt") SubtitleWriter::writeSrt(entries, out);
                else if (fmt == "vtt") SubtitleWriter::writeVtt(entries, out);
                else if (fmt == "txt") SubtitleWriter::writeTxt(entries, out);
                written << QString::fromStdString(out.string());
            }

            const auto entriesCount = static_cast<int>(entries.size());

            QMetaObject::invokeMethod(this, [this, entriesCount, written]() {
                appendLog(QString("Transcribed %1 entries.").arg(entriesCount));
                for (const auto& w : written) appendLog("Wrote: " + w);
                onExtractFinished(true,
                    tr("Done - %1 entries in %2 file(s).")
                        .arg(entriesCount).arg(written.size()));
            }, Qt::QueuedConnection);

        } catch (const std::exception& ex) {
            const QString msg = QString::fromStdString(ex.what());
            QMetaObject::invokeMethod(this, [this, msg]() {
                appendLog("ERROR: " + msg);
                onExtractFinished(false, tr("Failed."));
            }, Qt::QueuedConnection);
        }
    });
}
